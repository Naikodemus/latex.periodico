% NOTICIAS
\newcommand{\titulonoticia}[3]{%
	\ifthenelse{\equal{#1}{}}
		{}
		{{\raggedright\rmfamily\huge\textbf{#1}\par}}
	\ifthenelse{\equal{#2}{}}
		{\vspace{#3mm}}
		{{\raggedright\rmfamily\Large{\color{DarkGray} #2}\par\vspace{#3mm}}}%
}
